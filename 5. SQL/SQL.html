# чтобы выполнить SQL-запрос, пишем функцию с использованием Pandas
def get_sql_data(query:str, engine:sa.engine.base.Engine=engine) -> pd.DataFrame:
    '''Открываем соединение, получаем данные из sql, закрываем соединение'''
    with engine.connect() as con:
        return pd.read_sql(sql=sa.text(query), con = con)

reviews_rating = '''

SELECT b.title AS title,
       b.book_id AS id,
       AVG(r.rating) AS avg_rate,
       COUNT(DISTINCT rev.text) AS cnt_reviews
FROM   books AS b
INNER JOIN ratings AS r ON r.book_id = b.book_id
LEFT JOIN reviews AS rev ON rev.book_id = b.book_id
GROUP BY b.title,
         b.book_id
ORDER BY COUNT(DISTINCT rev.text) DESC
'''

       AVG(rating) AS avg_rating
FROM
    (SELECT author_id, 
            AVG(rating) AS rating
     FROM books as b
     JOIN ratings as r ON b.book_id = r.book_id
     GROUP BY author_id, 
              title
     HAVING COUNT(title) >= 50) AS req
JOIN authors AS a ON req.author_id = a.author_id
GROUP BY name_author
ORDER BY avg_rating DESC

users = '''

SELECT ROUND(AVG(count))
FROM
    (SELECT COUNT(text)
     FROM reviews
     FULL OUTER JOIN (SELECT username,
                      COUNT(rating_id) as cnt_rat
                      FROM ratings
                      GROUP BY username) AS cnt_rating
     ON cnt_rating.username = reviews.username
     where cnt_rat > 48
     GROUP BY reviews.username) AS cnt

'''

       COUNT(book_id) AS books_cnt,

top7 = '''

WITH 
users_pages AS
  (SELECT username,
          SUM(num_pages) AS pages_all
   FROM
     (SELECT username,
             book_id
      FROM reviews
      UNION
      SELECT username,
             book_id
             FROM ratings) AS users
    LEFT JOIN books as b ON users.book_id = b.book_id
    GROUP BY username),

users_rat_rev AS
(SELECT rev.username,
        COUNT(DISTINCT review_id) AS reviews_cnt,
        COUNT(DISTINCT rating_id) AS ratings_cnt
 FROM reviews AS rev
 JOIN ratings AS rat ON rev.username = rat.username
 GROUP BY rev.username)
 
SELECT up.username,
       reviews_cnt,
       ratings_cnt,
       pages_all
       
FROM users_pages AS up
JOIN users_rat_rev AS urr ON up.username = urr.username
ORDER BY reviews_cnt DESC, ratings_cnt DESC
LIMIT 7
'''
